! Topologia: ISP=Hub, R1-S1=Spoke1, R3=Spoke2
!
! Tabla de Direccionamiento:
! -------------------------------------------------------------------
! Dispositivo | Interfaz | IP               | Descripcion
! -------------------------------------------------------------------
! ISP (Hub)   | e0/0     | 20.25.1.1/30     | WAN hacia R1-S1
! ISP (Hub)   | Tunnel0  | 14.2.10.1/24     | DMVPN mGRE Hub
! R1-S1 (Sp1) | e0/0     | 20.25.1.2/30     | WAN hacia ISP
! R1-S1 (Sp1) | e0/2     | 20.25.2.5/30     | WAN hacia R3
! R1-S1 (Sp1) | Tunnel0  | 14.2.10.2/24     | DMVPN GRE Spoke1
! R3 (Sp2)    | e0/0     | 20.25.2.6/30     | WAN hacia R1-S1
! R3 (Sp2)    | e0/1     | 30.30.30.1/24    | LAN
! R3 (Sp2)    | Tunnel0  | 14.2.10.3/24     | DMVPN GRE Spoke2
! -------------------------------------------------------------------
! R3 llega al ISP (Hub) pasando por R1-S1 como transit
! PSK: Cisco123 | NHRP Auth: NHRP2024 | EIGRP AS 100
! ============================================================


! ============================================================
! ISP - HUB DMVPN
! ============================================================
enable
conf t
hostname ISP
!
! --- IKEv1 ---
crypto isakmp policy 10
 encr aes 256
 hash sha256
 authentication pre-share
 group 14
 lifetime 86400
!
crypto isakmp key Cisco123 address 0.0.0.0 0.0.0.0
!
crypto ipsec transform-set TS_DMVPN esp-aes 256 esp-sha256-hmac
 mode transport
!
crypto ipsec profile DMVPN_PROFILE
 set transform-set TS_DMVPN
!
interface Tunnel0
 ip address 14.2.10.1 255.255.255.0
 no ip redirects
 ip nhrp authentication NHRP2024
 ip nhrp map multicast dynamic
 ip nhrp network-id 100
 ip nhrp holdtime 300
 tunnel source e0/0
 tunnel mode gre multipoint
 tunnel protection ipsec profile DMVPN_PROFILE
!
interface e0/0
 ip address 20.25.1.1 255.255.255.252
 no shutdown
!
router eigrp 100
 network 14.2.10.0 0.0.0.255
 network 20.25.1.0 0.0.0.3
 no auto-summary
!
end


! ============================================================
! R1-S1 - SPOKE 1
! (Conecta directo al ISP; tambien es transit para R3)
! ============================================================
enable
conf t
hostname R1-S1
!
crypto isakmp policy 10
 encr aes 256
 hash sha256
 authentication pre-share
 group 14
 lifetime 86400
!
crypto isakmp key Cisco123 address 0.0.0.0 0.0.0.0
!
crypto ipsec transform-set TS_DMVPN esp-aes 256 esp-sha256-hmac
 mode transport
!
crypto ipsec profile DMVPN_PROFILE
 set transform-set TS_DMVPN
!
interface Tunnel0
 ip address 14.2.10.2 255.255.255.0
 ip nhrp authentication NHRP2024
 ip nhrp map multicast 20.25.1.1
 ip nhrp map 14.2.10.1 20.25.1.1
 ip nhrp network-id 100
 ip nhrp holdtime 300
 ip nhrp nhs 14.2.10.1
 tunnel source e0/0
 tunnel mode gre multipoint
 tunnel protection ipsec profile DMVPN_PROFILE
!
interface e0/0
 ip address 20.25.1.2 255.255.255.252
 no shutdown
!
interface e0/2
 ip address 20.25.2.5 255.255.255.252
 no shutdown
!
router eigrp 100
 network 14.2.10.0 0.0.0.255
 network 20.25.1.0 0.0.0.3
 network 20.25.2.4 0.0.0.3
 no auto-summary
!
end


! ============================================================
! R3 - SPOKE 2
! (Llega al ISP via R1-S1 como transit; tunnel source = e0/0)
! ============================================================
enable
conf t
hostname R3
!
crypto isakmp policy 10
 encr aes 256
 hash sha256
 authentication pre-share
 group 14
 lifetime 86400
!
crypto isakmp key Cisco123 address 0.0.0.0 0.0.0.0
!
crypto ipsec transform-set TS_DMVPN esp-aes 256 esp-sha256-hmac
 mode transport
!
crypto ipsec profile DMVPN_PROFILE
 set transform-set TS_DMVPN
!
! NHRP apunta al Hub ISP (20.25.1.1); R1-S1 ruteara el GRE
interface Tunnel0
 ip address 14.2.10.3 255.255.255.0
 ip nhrp authentication NHRP2024
 ip nhrp map multicast 20.25.1.1
 ip nhrp map 14.2.10.1 20.25.1.1
 ip nhrp network-id 100
 ip nhrp holdtime 300
 ip nhrp nhs 14.2.10.1
 tunnel source e0/0
 tunnel mode gre multipoint
 tunnel protection ipsec profile DMVPN_PROFILE
!
interface e0/0
 ip address 20.25.2.6 255.255.255.252
 no shutdown
!
interface e0/1
 ip address 30.30.30.1 255.255.255.0
 no shutdown
!
! Ruta al ISP via R1-S1
ip route 20.25.1.0 255.255.255.252 20.25.2.5
ip route 14.2.10.0 255.255.255.0 20.25.2.5
!
router eigrp 100
 network 14.2.10.0 0.0.0.255
 network 20.25.2.4 0.0.0.3
 network 30.30.30.0 0.0.0.255
 no auto-summary
!
end
